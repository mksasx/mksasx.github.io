   public static void bubbleSort(int[] args) {
        int len = args.length;
        while (len > 0) {
            for (int i = 0; i < len - 1; i++) {
                int next = i + 1;
                if (args[i] > args[next]) {
                    int temp = args[next];
                    args[next] = args[i];
                    args[i] = temp;
                }
            }
            len--;
        }
    }
